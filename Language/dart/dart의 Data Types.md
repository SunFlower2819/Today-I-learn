### 🤞List

```dart
void main() {
  var addNumFive = true;
  
  List<int> int_numbers = [
    1,
    2,
    3,
    if(addNumFive) 5,  // 만약 addNumFive가 true면 리스트에 5를 추가해라
  ];
  
  var double_numbers = [3.13, 13.13, 4.1];  // List<double> 타입
  
  print(int_numbers);      // [1,2,3,5] 출력
  print(double_numbers); 
}
```
